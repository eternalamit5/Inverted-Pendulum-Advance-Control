\section{Linearisation}

\begin{align}
	\ddot{\alpha}=\dfrac{3}{2l}\left(g\sin(\alpha)-\ddot{x}_w\cos(\alpha) \right) 
\end{align}

\begin{align}
	f_{\text{lin}}=f(x_0)+\dfrac{\partial f}{\partial x}(x_0)\cdot\left(x-x_0 \right) +\underbrace{r(x)}\\
	\quad \quad \quad \approx 0 \nonumber
\end{align}

\begin{align}
	\ddot{\alpha}&=\dfrac{3}{2l}\left[ g\sin(x_0)-\ddot{x}_w\cos(x_0)+\left( g\cos(x_0)+\ddot{x}_w\sin(x_0)\right)\cdot\left(\alpha-x_0 \right) \right] \qquad\Bigg|x_0=0 \\ \nonumber
	&=\dfrac{3}{2l}\left[ g\sin(0)-\ddot{x}_w\cos(0)+\left( g\cos(0)+\ddot{x}_w\sin(0)\right)\cdot\left(\alpha-0 \right) \right]\\ \nonumber
	&=\dfrac{3}{2l}\left(g\alpha  -\ddot{x}_w \right)
\end{align}

\begin{align}
	\ddot{x}_w&=-\dfrac{1}{T_1}\dot{x}_w+\dfrac{1}{T_1}u \nonumber\\
	\ddot{\alpha}&=\dfrac{3}{2l}\left(g\alpha  -\ddot{x}_w \right)\\
	\Rightarrow\ddot{\alpha}&=\dfrac{3}{2l}\left(g\alpha + \dfrac{1}{T_1}\dot{x}_w-\dfrac{1}{T_1}u \right)\\
\end{align}

\begin{align}
	\begin{pmatrix}
	\dot{x}_w \\
	\ddot{x}_w \\
	\dot{\alpha}\\
	\ddot{\alpha}
	\end{pmatrix}
	=
	\begin{pmatrix}
	0&1&0&0\\
	0&-\frac{1}{T_1}&0&0\\
	0&0&0&1\\
	0&\frac{3}{2lT_1}&\frac{3g}{2l}&0
	\end{pmatrix}
	\begin{pmatrix}
	x_w \\
	\dot{x}_w \\
	{\alpha}\\
	\dot{\alpha}
	\end{pmatrix}
	+
	\begin{pmatrix}
	0\\
	\frac{1}{T_1}\\
	0\\
	-\frac{1}{T_1}
	\end{pmatrix}
	u
\end{align}

\begin{align}
	\begin{pmatrix}
	\dot{x}_w\\
	\dot{\alpha}
	\end{pmatrix}
	=
	\begin{pmatrix}
	0&1&0&0\\
	0&0&0&1
	\end{pmatrix}
	\begin{pmatrix}
	x_w \\
	\dot{x}_w \\

	\dot{\alpha}
	\end{pmatrix}
\end{align}
